Private Sub TAT_HPEcharts_groupage_Click()

Dim TAT_HPE_Gr As Workbook
Dim HPEchSheetgr, HPEchSheetdi As Worksheet
Dim sGrTable, sGrLanes, sGrLanesCell, GrTATcell, TATShName As range
Dim TATShNameString As String


Set SourceGR = Workbooks(Dir("C:\Users\Rzeglarz\Desktop\Weekly\source*.xlsb")).Worksheets("groupage")
Set TAT_HPE_Gr = Workbooks("HPE_TAT_groupage_chart.xlsx")
Set sGrTable = SourceGR.UsedRange
Set sGrLanes = SourceGR.range(SourceGR.range("a2"), SourceGR.range("a2").End(xlDown))

WeekNo = Format(Date, "ww")


i = 1
    For Each sGrLanesCell In sGrLanes
        sGrLanesCell.NumberFormat = "@"
        sGrLanesCell.value = Left(sGrLanesCell, 2) & " " & "-" & " " & Right(sGrLanesCell, 2)
        i = i + 1
    Next
    
Set sGrLanesCell = sGrLanes.cells

i = 1
For Each HPEchSheetgr In TAT_HPE_Gr.Worksheets
    

    HPEchSheetgr.Activate
    lastrow = ActiveSheet.cells(ActiveSheet.Rows.Count, "K").End(xlUp).Row
   
        If InStr(1, ActiveSheet.Name, sGrLanesCell(i).value) Then
    
            If lastrow = 21 Then
                HPEchSheetgr.range("K3:Q21").Copy Destination:=HPEchSheetgr.range("K3").Offset(-1)
                HPEchSheetgr.range("K21", "Q21").value = vbNullString
                HPEchSheetgr.range("K21").value = "2017_0" & WeekNo - 1
                sGrLanesCell(i).Offset(0, 2).Resize(1, 6).Copy
                HPEchSheetgr.range("L21", "Q21").PasteSpecial (xlPasteValues)
                Else
                ActiveSheet.cells(lastrow, "K").Offset(1, 0).value = "2017_0" & WeekNo - 1
                ActiveSheet.cells(lastrow + 1, "K").Offset(0, 1).Resize(1, 6).value = vbNullString
                sGrLanesCell(i).Offset(0, 2).Resize(1, 6).Copy
                ActiveSheet.cells(lastrow + 1, "K").Offset(0, 1).PasteSpecial (xlPasteValues)
                
            End If
            i = i + 1
        End If

Next


End Sub
