Function findOpenTATcharts_HPInc()
Dim wrk As Workbook

    For Each wrk In Workbooks
        If InStr(wrk.Name, "TAT Segmentation version") > 0 Then
            wrk.Worksheets(1).Activate
            Exit For
        End If
    Next
End Function

Function findOpenSource_HPInc()
Dim wrk As Workbook

    For Each wrk In Workbooks
        If InStr(wrk.Name, "HPI_source") > 0 Then
            wrk.Worksheets("overall").Activate
            Exit For
        End If
    Next
End Function
